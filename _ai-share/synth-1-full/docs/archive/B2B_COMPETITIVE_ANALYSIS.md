### **УРОВЕНЬ 1: КРИТИЧНО (Фаза 3)**

#### 1. **UNIFIED USER PROFILE SYNC** 🔄
**Проблема**: Сейчас профиль пользователя (`UserProfile`) и B2B-дашборд не синхронизированы.

**Решение**:
```typescript
// src/app/shop/b2b/page.tsx
const { user } = useAuth();
const { organizations, activeOrganizationId, roles } = user || {};

// Определение контекста
const currentOrg = organizations?.find(o => o.organizationId === activeOrganizationId);
const isRetailer = currentOrg?.roleInOrg === 'admin' && roles?.includes('shop');
const isBrand = currentOrg?.roleInOrg === 'admin' && roles?.includes('brand');
const isBuyer = currentOrg?.roleInOrg === 'member' && 
                user?.professionalRoles?.includes('buyer');
const isSalesRep = user?.professionalRoles?.includes('sales_representative');
```

**Фичи**:
- **Organization Switcher** в хедере (если пользователь в нескольких org)
- **Role Badge** показывает текущую роль (Brand Manager / Buyer / Sales Rep)
- **Permission-based UI** - скрытие недоступных функций
- **Activity Context** - логи показывают organization context

**Дизайн**:
```tsx
<header>
  <OrganizationSwitcher 
    orgs={user.organizations}
    active={user.activeOrganizationId}
    onChange={switchOrg}
  />
  <RoleBadge role={currentOrg.roleInOrg} />
  <PermissionIndicator permissions={currentOrg.permissions} />
</header>
```

**Метрики**: +60% ясности контекста, -40% ошибок доступа

```tsx
          </p>
        </div>
        <Button size="sm">Add to Cart</Button>
      </div>
    </div>
  </CardContent>
</Card>
```

**Алгоритм**:
```typescript
function calculateReorderPoint(sku: string) {
  const sellRate = getSellThroughRate(sku); // units per day
  const leadTime = getSupplierLeadTime(sku); // days
  const safetyStock = sellRate * 7; // 1 week buffer
  const reorderPoint = (sellRate * leadTime) + safetyStock;
  
  if (currentStock < reorderPoint) {
    return {
      urgency: 'high',
      daysUntilStockout: currentStock / sellRate,
      suggestedQuantity: calculateEOQ(sku)
    };
  }
}
```

```typescript
function detectPrimaryRole(
  user: UserProfile, 
  org: Organization
): string {
  // Логика определения главной роли
  const orgAccess = user.organizations?.find(o => o.organizationId === org.id);
  const isOrgAdmin = orgAccess?.roleInOrg === 'admin';
  const hasBuyerRole = user.professionalRoles?.includes('buyer');
  const hasSalesRole = user.professionalRoles?.includes('sales_representative');
  
  if (isOrgAdmin && org.type === 'shop') return 'retailer_owner';
  if (isOrgAdmin && org.type === 'brand') return 'brand_manager';
  if (hasBuyerRole) return 'buyer';
  if (hasSalesRole) return 'sales_rep';
  
  return 'default';
}
```

```typescript
// src/components/dashboard/FeatureGate.tsx

export function FeatureGate({ 
  feature, 
  user, 
  org, 
  children 
}: FeatureGateProps) {
  const orgAccess = user.organizations?.find(o => o.organizationId === org.id);
  const permissions = orgAccess?.permissions;
  
  const featurePermissions: Record<string, keyof PermissionSet> = {
    'finance_dashboard': 'canViewFinances',
    'team_management': 'canManageTeam',
    'analytics': 'canViewAnalytics',
    'order_approval': 'canApproveOrders',
    'audit_trail': 'canAccessAuditTrail'
  };
  
  const requiredPermission = featurePermissions[feature];
  const hasPermission = permissions?.[requiredPermission];
  
  if (!hasPermission) {
    return (
      <Card className="opacity-50">
        <CardContent className="p-8 text-center">
          <Lock className="h-12 w-12 text-slate-300 mx-auto mb-4" />
          <h3 className="font-bold text-slate-900 mb-2">Access Restricted</h3>
          <p className="text-sm text-slate-500">
            Contact your admin to request {feature} access
          </p>
        </CardContent>
      </Card>
    );
  }
  
  return <>{children}</>;
}
```
